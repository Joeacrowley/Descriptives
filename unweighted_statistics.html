5.2.1 unweighted_mean

unweighted_mean <- function(
    data, # a data frame
    outcomes, # a character vector of outcome variables
    predictors = NULL, # a list, each element containing a character vector of predictor variables
    conf = NULL, # "se" or "ci"; returns either standard error or confidence intervals 
    base = NULL
) {suppressMessages({suppressWarnings({
  
  output <- map(outcomes, function(cur_outcome) {
    
    base_defintion <- ifelse(is.null(base), NA, create_bases(base_info = base, variables = cur_outcome)) 
        
    # Remove missing data
    one_total_table <- data %>% 
      filter(!is.na(.data[[cur_outcome]])) %>% 
      summarise(estimate = mean(!!sym(cur_outcome), na.rm = T), 
          n = n(), 
          sd = sd(!!sym(cur_outcome), na.rm = T), 
          estimate_se = sd / sqrt(n), 
          estimate_low = estimate - 1.96 * estimate_se,
          estimate_upp = estimate + 1.96 * estimate_se, 
          base = sum(n)) %>% 
          select(-n, -sd) %>% 
        mutate("{cur_outcome}" := "mean") %>% 
        mutate(outcome = cur_outcome) %>% 
        rename(any_of(c(o_cat = cur_outcome))) %>% 
        mutate(o_lab = attributes(data[[cur_outcome]])$label) %>% 
        mutate(base_description = base_defintion) # descriptor label for base 
  
    # Create cross-breaks
    if(!is.null(predictors)){
    predictor_tables <- map(predictors, function(cur_predictor) { 
      
            # Handle cases with and without predictor variables
        if (length(cur_predictor) > 0) {
          
          # Vector containing predictors and outcomes
          predictor_and_outcome <- c(cur_predictor, cur_outcome)
          base_defintion <- ifelse(is.null(base), NA, create_bases(base_info = base, variables = predictor_and_outcome)) 
          
          one_predictor_table <- data %>%
            # Filter missing values
            filter(rowSums(across(all_of(predictor_and_outcome), ~ is.na(.))) == 0) %>%
            # Group data
            group_by(across(all_of(cur_predictor))) %>% 
            # Weighted mean + base sizes, with grouping vars
            summarise(estimate = mean(!!sym(cur_outcome), na.rm = T), 
                      n = n(), 
                      sd = sd(!!sym(cur_outcome), na.rm = T), 
                      estimate_se = sd / sqrt(n), 
                      estimate_low = estimate - 1.96 * estimate_se,
                      estimate_upp = estimate + 1.96 * estimate_se, 
                      base = sum(n)) %>% 
                      select(-n, -sd) %>% 
            # Added for consistency when merged with categorical outcome tables. 
            mutate("{cur_outcome}" := "mean") %>% 
            # Standardise names so can stack results across different predictors and outcomes. 
            standardise_names(data = data, preds = cur_predictor, out_var = cur_outcome) %>%
            mutate(cross_break = paste0(cur_predictor, collapse = "_X_")) %>%
            mutate(base_description = base_defintion) # descriptor label for base 
        }
      
      return(one_predictor_table)
      
    })
    
    # Stack tables one on another
    predictor_tables <- bind_rows(predictor_tables)
    
    } # Predictors ends
    
    # Bind crossbreaks and frequency table together
    if(!is.null(predictors)){table_for_one_outcome <- bind_rows(one_total_table, predictor_tables)
    } else {table_for_one_outcome <- one_total_table}
    
    table_for_one_outcome <- table_for_one_outcome %>%
      mutate(stat = "mean") %>%
      # Add 'Total' to predictor category variables for frequency section of the table.
      mutate(across(contains(c("cross_break", "predictor", "p_lab", "p_cat")), ~case_when(is.na(.x) ~ "Total", TRUE ~ as.character(.x)))) %>%
      # Reorder variables
      select(contains(c("cross_break", "predictor", "p_lab", "p_cat")), outcome, o_lab, o_cat, stat, contains("estimate"), base, base_description) 
  
    if(!is.null(conf)){
      if(conf == "ci"){
        table_for_one_outcome <- table_for_one_outcome %>% 
          mutate(estimate_ci = paste0(as.character(estimate_low), " - ", as.character(estimate_upp)))}
    } else { 
      table_for_one_outcome <- table_for_one_outcome %>% mutate(estimate_se = "-", .after = estimate) %>% 
        select(-estimate_low, -estimate_upp)
    }
    
    return(table_for_one_outcome)
    
  }) # Outcomes loop ends 
  
  # Stack tables one on another
  output <- output %>% bind_rows() %>% mutate(unweighted_n = base)
  
  # Either drop standard errors or confidence intervals
  if(!is.null(conf)){
  if(conf == "se"){keep <- c("upp", "low")} else {keep <- "_se"}
    output <- output %>% select(!ends_with(keep))
  }
  
  return(output)
  
})})
}

unweighted_median <- function(
    data, # a data frame
    outcomes, # a character vector of outcome variables
    predictors = NULL, # a list, each element containing a character vector of predictor variables

  output <- map(outcomes, function(cur_outcome) {
    
    base_defintion <- ifelse(is.null(base), NA, create_bases(base_info = base, variables = cur_outcome)) 
        
    # Remove missing data
    one_total_table <- data %>% 
      filter(!is.na(.data[[cur_outcome]])) %>% 
      summarise(estimate = median(!!sym(cur_outcome), na.rm = T),
                base = n())
      
    if(!is.null(conf)){
      if(conf == "se"){
      median_se <- wrappedtools::medianse(data %>% pull(cur_outcome))
      one_total_table <- one_total_table %>% mutate(estimate_se = median_se)
      } else {
      median_ci <- wrappedtools::median_cl_boot(data %>% pull(cur_outcome))
      one_total_table <- one_total_table %>% mutate(
        estimate_low = median_ci$CIlow,
        estimate_upp = median_ci$CIhigh)
      }
    } else {
      one_total_table <- one_total_table %>% mutate(estimate_se = "-")
    }
    
     one_total_table <- one_total_table %>% 
        mutate("{cur_outcome}" := "median") %>% 
        mutate(outcome = cur_outcome) %>% 
        rename(any_of(c(o_cat = cur_outcome))) %>%

5.2.3 unweighted_sum

unweighted_sum <- function(
    data, # a data frame
    outcomes, # a character vector of outcome variables
    predictors = NULL, # a list, each element containing a character vector of predictor variables
    conf = NULL, # "se" or "ci"; returns either standard error or confidence intervals 
    base = NULL
) {suppressMessages({suppressWarnings({
  
  output <- map(outcomes, function(cur_outcome) {
    
    base_defintion <- ifelse(is.null(base), NA, create_bases(base_info = base, variables = cur_outcome)) 

    # Remove missing data
    one_total_table <- data %>% 
      filter(!is.na(.data[[cur_outcome]])) %>% 
      summarise(estimate = sum(!!sym(cur_outcome), na.rm = T), 
            base = n(), 
            sd = sd(!!sym(cur_outcome), na.rm = T), 
            estimate_se = sd * sqrt(base), 
            estimate_low = estimate - 1.96 * estimate_se,
            estimate_upp = estimate + 1.96 * estimate_se
            ) %>% select(-sd)
    
     one_total_table <- one_total_table %>% 
        mutate("{cur_outcome}" := "sum") %>% 
        mutate(outcome = cur_outcome) %>% 
        rename(any_of(c(o_cat = cur_outcome))) %>% 
        mutate(o_lab = attributes(data[[cur_outcome]])$label) %>% 
        mutate(base_description = base_defintion) # descriptor label for base 
  
    # Create cross-breaks
    if(!is.null(predictors)){
    predictor_tables <- map(predictors, function(cur_predictor) { 
      
            # Handle cases with and without predictor variables
        if (length(cur_predictor) > 0) {
          
          # Vector containing predictors and outcomes
          predictor_and_outcome <- c(cur_predictor, cur_outcome)
          base_defintion <- ifelse(is.null(base), NA, create_bases(base_info = base, variables = predictor_and_outcome)) 
          
          one_predictor_table <- data %>%
            # Filter missing values
            filter(rowSums(across(all_of(predictor_and_outcome), ~ is.na(.))) == 0) %>%
            # Group data
            group_by(across(all_of(cur_predictor))) %>% 
            # Weighted mean + base sizes, with grouping vars
            summarise(estimate = sum(!!sym(cur_outcome), na.rm = T), 
            base = n(), 
            sd = sd(!!sym(cur_outcome), na.rm = T), 
            estimate_se = sd * sqrt(base), 
            estimate_low = estimate - 1.96 * estimate_se,
            estimate_upp = estimate + 1.96 * estimate_se
            ) %>% select(-sd)
      
          # Added for consistency when merged with categorical outcome tables. 
          one_predictor_table <- one_predictor_table %>% mutate("{cur_outcome}" := "sum") %>% 
            # Standardise names so can stack results across different predictors and outcomes. 
            standardise_names(data = data, preds = cur_predictor, out_var = cur_outcome) %>%
            mutate(cross_break = paste0(cur_predictor, collapse = "_X_")) %>%
            mutate(base_description = base_defintion) # descriptor label for base 
            
        }
      
      return(one_predictor_table)
      
    })
    
    # Stack tables one on another
    predictor_tables <- bind_rows(predictor_tables)
    
    } # Predictors ends
    
    # Bind crossbreaks and frequency table together
    if(!is.null(predictors)){table_for_one_outcome <- bind_rows(one_total_table, predictor_tables)
    } else {table_for_one_outcome <- one_total_table}
    
    table_for_one_outcome <- table_for_one_outcome %>%
      mutate(stat = "sum") %>% 
      # Add 'Total' to predictor category variables for frequency section of the table.
      mutate(across(contains(c("cross_break", "predictor", "p_lab", "p_cat")), ~case_when(is.na(.x) ~ "Total", TRUE ~ as.character(.x)))) %>%
      # Reorder variables
      select(contains(c("cross_break", "predictor", "p_lab", "p_cat")), outcome, o_lab, o_cat, stat, contains("estimate"), base, base_description) 
    
    if(!is.null(conf)){
      if(conf == "ci"){
        table_for_one_outcome <- table_for_one_outcome %>% 
          mutate(estimate_ci = paste0(as.character(estimate_low), " - ", as.character(estimate_upp)))
      }
    } else {
      table_for_one_outcome <- table_for_one_outcome %>% mutate(estimate_se = "-") %>% 
        select(-estimate_upp, -estimate_low)
    }
    
    return(table_for_one_outcome)
    
  }) # Outcomes loop ends 
  
  # Stack tables one on another
  output <- output %>% bind_rows() %>% mutate(unweighted_n = base)
  
  # Either drop standard errors or confidence intervals
  if(!is.null(conf)){
  if(conf == "se"){keep <- c("upp", "low")} else {keep <- "_se"}
    output <- output %>% select(!ends_with(keep))
  }
  
  return(output)
  
})})
}

write_function(unweighted_sum, folder = "Scripts")

unweighted_perc <- function(
    data, # a data frame
    outcomes, # a character vector of outcome variables
    predictors = NULL, # a list, each element containing a character vector of predictor variables
    conf = NULL, # "se" or "ci"; returns either standard error or confidence intervals 
    base = NULL
) {suppressMessages({suppressWarnings({
  
  output <- map(outcomes, function(cur_outcome) {
    
    base_defintion <- ifelse(is.null(base), NA, create_bases(base_info = base, variables = cur_outcome))

    # Remove missing data
    one_total_table <- data %>% 
      filter(!is.na(.data[[cur_outcome]])) %>% 
      count(.data[[cur_outcome]], name = "unweighted_n") %>%    
      mutate(estimate = unweighted_n / sum(unweighted_n), 
             estimate_se = sqrt(estimate * (1 - estimate) / sum(unweighted_n)), 
             estimate_low = (estimate - 1.96 * estimate_se),
             estimate_upp = (estimate + 1.96 * estimate_se),
             base = sum(unweighted_n, na.rm = T)
             ) %>% 
        mutate(outcome = cur_outcome) %>% 
        rename(any_of(c(o_cat = cur_outcome))) %>% 
        mutate(o_lab = attributes(data[[cur_outcome]])$label) %>% 
        mutate(base_description = base_defintion) # descriptor label for base 
  
    # Create cross-breaks
    if(!is.null(predictors)){
    predictor_tables <- map(predictors, function(cur_predictor) { 
      
            # Handle cases with and without predictor variables
        if (length(cur_predictor) > 0) {
          
          # Vector containing predictors and outcomes
          predictor_and_outcome <- c(cur_predictor, cur_outcome)
          base_defintion <- ifelse(is.null(base), NA, create_bases(base_info = base, variables = predictor_and_outcome)) 
          
          one_predictor_table <- data %>%
            # Filter missing values
            filter(rowSums(across(all_of(predictor_and_outcome), ~ is.na(.))) == 0) %>%
            # Group data
            group_by(across(all_of(predictor_and_outcome))) %>% 
            summarise(unweighted_n = n()) %>%
            group_by(across(all_of(cur_predictor))) %>% 
            mutate(estimate = unweighted_n / sum(unweighted_n, na.rm = T), 
               estimate_se = sqrt(estimate * (1 - estimate) / sum(unweighted_n)), 
               estimate_low = (estimate - 1.96 * estimate_se),
               estimate_upp = (estimate + 1.96 * estimate_se),
               base = sum(unweighted_n, na.rm = T)
             )
      
          # Added for consistency when merged with categorical outcome tables. 
          one_predictor_table <- one_predictor_table %>% 
            # Standardise names so can stack results across different predictors and outcomes. 
            standardise_names(data = data, preds = cur_predictor, out_var = cur_outcome) %>%
            mutate(cross_break = paste0(cur_predictor, collapse = "_X_")) %>%
            mutate(base_description = base_defintion) # descriptor label for base 
            
        }
      
      return(one_predictor_table)
      
    })
    
    # Stack tables one on another
    predictor_tables <- bind_rows(predictor_tables)
    
    } # Predictors ends
    
    # Bind crossbreaks and frequency table together
    if(!is.null(predictors)){table_for_one_outcome <- bind_rows(one_total_table, predictor_tables)
    } else {table_for_one_outcome <- one_total_table}
    
    table_for_one_outcome <- table_for_one_outcome %>%
      mutate(stat = "perc") %>%
      # Add 'Total' to predictor category variables for frequency section of the table.
      mutate(across(contains(c("cross_break", "predictor", "p_lab", "p_cat")), ~case_when(is.na(.x) ~ "Total", TRUE ~ as.character(.x)))) %>%
      # Reorder variables
      select(contains(c("cross_break", "predictor", "p_lab", "p_cat")), outcome, o_lab, o_cat, stat, contains("estimate"), unweighted_n, base, base_description) 
    
    if(!is.null(conf)){
      if(conf == "ci"){
        table_for_one_outcome <- table_for_one_outcome %>% 
          mutate(estimate_ci = paste0(as.character(estimate_low), " - ", as.character(estimate_upp)))
      }
    } else {
      table_for_one_outcome <- table_for_one_outcome %>% mutate(estimate_se = "-") %>% 
        select(-estimate_upp, -estimate_low)
    }
    

    return(table_for_one_outcome)
    
  }) # Outcomes loop ends 
  
  # Stack tables one on another
  output <- output %>% bind_rows()
  
  # Either drop standard errors or confidence intervals
  if(!is.null(conf)){
  if(conf == "se"){keep <- c("upp", "low")} else {keep <- "_se"}
    output <- output %>% select(!ends_with(keep))
  }
  
  return(output)
  
})})
}
