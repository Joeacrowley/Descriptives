make_hux <- function(table){
  
  (predictor_label <- table %>% select(p_lab1) %>% filter(p_lab1 != "Total") %>% unlist %>% unique)
  
  (n_predictor_levels <- table %>% select(p_cat1) %>% filter(p_cat1 != "Total") %>% unlist %>% unique %>% length)
  
  (preds_df <- tibble(var = rep(predictor_label, n_predictor_levels)) %>% t() %>% as.data.frame() %>% unname)

  ht <- as_hux(preds_df, add_colnames = FALSE)
  ht <- merge_cells(ht, 1, 1:n_predictor_levels)
  ht <- insert_column(ht, "", after = 0)
  ht <- insert_column(ht, "", after = 0)
  
  return(ht)
  
}
