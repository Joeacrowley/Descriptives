This function then returns list depth. If depth equals 1 then use same predictors for all outcomes. If depth equals 2 then try and use different sets of predictors for different outcomes.

list_depth <- function(x) {
  if (!is.list(x)) {
    return(0)  # Not a list → depth 0
  } else if (length(x) == 0) {
    return(1)  # Empty list → depth 1
  } else {
    return(1 + max(sapply(x, list_depth)))
  }
}

write_function(list_depth, folder = "Scripts")

# Returns TRUE if all variables in list appear in data
# Otherwise returns FALSE
vars_exist <- function(variable_list, data){
  
  if(any(grepl("survey", class(data)))){
    df_to_check <- data[["variables"]]
    } else {
      df_to_check <- data
      }
  
  predictors <- variable_list %>% unlist %>% unique
  variables_in_df <- names(df_to_check)
  
  variables_in_both <- intersect(predictors, variables_in_df)
  
  length(predictors) == length(variables_in_both)

  }

check_all_factors <- function(data, variable_list){

  if(any(grepl("survey", class(data)))){
    df_to_check <- data[["variables"]]
    } else {
      df_to_check <- data
      }
  
  df_to_check %>% select(all_of(unlist(variable_list))) %>%
    map_chr(., ~paste0(class(.x), collapse = ", ")) %>% 
    unique == "factor"
}

outcomes_not_in_predictors <- function(outcomes, predictors){

  predictor_vector <- unlist(predictors) %>% unique
  overlap_of_predictors_and_outcomes <- intersect(outcomes, predictor_vector)
  length(overlap_of_predictors_and_outcomes) == 0
  
}

7.3 Wrapper to combine things - calc_stats

Checks:

  • Confirm outcomes not in list of predictors outcomes_not_in_predictors.
calc_stats <- function(
    data,
    outcomes, # Character vector of outcome variables
    predictors = NULL, # a list, usually each element contains a character vector of predictor variables
    statistics = c("count","mean"),  # default, unweighted count or mean
    conf = NULL, # "se" or "ci"
    base = NULL, 
    pval = NULL, # Whether to compute p-values (if not NULL)
    multicode = TRUE
    ){
  
  # Correct common, non-harmful, and inevitable mistakes #######################
  
  # If `predictors` is a character vector put it in a list
  if(class(predictors) == "character"){predictors <- list(predictors)}
  
  
  # Error handling #############################################################

  # Check variables do, in fact, exist. 
  vector_of_variables_concerned <- c(outcomes, unlist(unique(predictors)))
  test <- vars_exist(data = data, variable_list = vector_of_variables_concerned)
  if (test != TRUE) stop("Some variables you want to use do not exist in the data frame.")

  if(!is.null(predictors)){
    
  # Check outcomes not in predictors
  test <- outcomes_not_in_predictors(outcomes = outcomes, predictors = predictors)
  if (test != TRUE) stop("Outcome appears in predictor list.")

  # Confirm predictors are factors
    test <- check_all_factors(data = data, variable_list = predictors)
    if (test != TRUE) stop("Not all predictors are factor variables. Make them factors.")
  
  # Check predictor list depth is 1
  test <- list_depth(predictors) == 1
  if (test != TRUE) stop("Predictor list should not contained further lists.")
  }
  
  suppressMessages({suppressWarnings({ 
    
  # Where necessary, extract unweighted data from survey objects  
  if(any(str_detect(class(data), "survey"))){
    unweighted_data <- data[["variables"]]
    weighted_data <- data
  } else {
      unweighted_data <- data
      weighted_data <- NULL
  }
    
  # Compute statistics for each outcome/predictor combination 
  individual_tables <- pmap(list(outcomes), function(cur_outcome){
    
    # Extract current outcome variable and filter statistics based on variable type
    single_outcome <- unweighted_data %>% pull(cur_outcome) 
    if(is.factor(single_outcome)){statistic <- str_remove_all(statistics, "mean|median|sum")}
    if(is.numeric(single_outcome)){statistic <- str_remove_all(statistics, "perc|count")}
    
    results <- list() # Store stat results for this combo

    if("mean" %in% statistic){results <- append(results, list(unweighted_mean(unweighted_data, outcomes = cur_outcome, predictors = predictors, conf = conf, base = base)))}
    if("median" %in% statistic){results <- append(results, list(unweighted_median(unweighted_data, outcomes = cur_outcome, predictors = predictors, conf = conf, base = base)))}
    if("sum" %in% statistic){results <- append(results, list(unweighted_sum(unweighted_data, outcomes = cur_outcome, predictors = predictors, conf = conf, base = base)))}
    if("perc" %in% statistic){results <- append(results, list(unweighted_perc(unweighted_data, outcomes = cur_outcome, predictors = predictors, conf = conf, base = base)))}
    if("count" %in% statistic){results <- append(results, list(unweighted_count(unweighted_data, outcomes = cur_outcome, predictors = predictors, conf = conf, base = base)))}
    
    if(!is.null(weighted_data)){
    if("w_mean" %in% statistic){results <- append(results, list(weighted_mean(weighted_data, outcomes = cur_outcome, predictors = predictors, conf = conf, base = base)))}
    if("w_median" %in% statistic){results <- append(results, list(weighted_median(weighted_data, outcomes = cur_outcome, predictors = predictors, conf = conf, base = base)))}
    if("w_sum" %in% statistic){results <- append(results, list(weighted_sum(weighted_data, outcomes = cur_outcome, predictors = predictors, conf = conf, base = base)))}
    if("w_perc" %in% statistic){results <- append(results, list(weighted_perc(weighted_data, outcomes = cur_outcome, predictors = predictors, conf = conf, base = base)))}
    if("w_count" %in% statistic){results <- append(results, list(weighted_count(weighted_data, outcomes = cur_outcome, predictors = predictors, conf = conf, base = base)))}
    }
    
    # Combine results for this outcome
    if(length(results) > 0){results <- results %>% reduce(bind_rows)}

    return(results)
  
  })

  # Merge all results into one table 
  merged_tables <- individual_tables %>% bind_rows
  
  if(!is.null(pval)){
    
    # Extract predictors eligible for p-value testing (only single variables)
    pvalue_preds <- predictors[predictors %>% map_lgl(., ~length(.x) == 1)]
    pvalue_preds <- pvalue_preds %>% unlist %>% unique
    
  pvalues <- list()
  
  # Unweighted p-values
  if(any(statistics %in% c("mean","median","sum","perc","count"))){
    unw_pvalues <- map_return_p_values(data = unweighted_data, outcomes  = outcomes, predictors = pvalue_preds)
    pvalues <- append(pvalues, list(unw_pvalues))
  }

  # Weighted p-values
  if(!is.null(weighted_data)){
    if(any(statistics %in% c("w_mean", "w_median", "w_sum", "w_perc", "w_count"))){
      wgt_pvalues <- map_return_p_values(data = data, outcomes  = outcomes, predictors = pvalue_preds)
      pvalues <- append(pvalues, list(wgt_pvalues))
      }
  }
  
  # Merge p-values into main results
  pvalues <- pvalues %>% bind_rows()
  merged_tables <- merged_tables %>% left_join(pvalues)
  
  } else { 
    # If no p-values requested, still add empty columns
    merged_tables <- merged_tables %>% mutate(p_method = NA, p_value = NA)
  }
  
  # Handle the case of no predictors (total summary) 
  if(is.null(predictors)){merged_tables <- merged_tables %>% mutate(cross_break = "Total")}
  
  # Convert multi-coded values using base info
  # Inserted if statement so that if mulitcodes functions not available, can be excluded. 
  if(multicode == T){
  merged_tables <- convert_multicodes(
    data = merged_tables,  
    base_info = base, 
    keep = "Yes"
    ) 
  }

  return(merged_tables)
  
})})
}

write_function(calc_stats, folder = "Scripts")
