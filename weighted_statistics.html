6.2.1 Weighted means

weighted_mean <- function(
    data, # a srvyr design object
    outcomes, # a character vector of outcome variables
    predictors = NULL, # a list, each element containing a character vector of predictor variables
    conf = NULL, # "se" or "ci"; returns either standard error or confidence intervals
    base = NULL
) {suppressMessages({suppressWarnings({
  
  output <- map(outcomes, function(cur_outcome) {
    
    base_defintion <- ifelse(is.null(base), NA, create_bases(base_info = base, variables = cur_outcome)) # edit
    
    # Remove missing data
    one_total_table <- data %>% 
      filter(!is.na(.data[[cur_outcome]])) %>% 
      summarise(estimate = survey_mean(!!sym(cur_outcome), vartype = conf), 
                base = unweighted(n())) %>% 
        mutate("{cur_outcome}" := "w_mean") %>% 
        mutate(outcome = cur_outcome) %>% 
        rename(any_of(c(o_cat = cur_outcome))) %>% 
        mutate(o_lab = attributes(data$variables[[cur_outcome]])$label) %>% 
      # edit
      mutate(base_description = base_defintion) # descriptor label for base 

    # Create cross-breaks
    if(!is.null(predictors)){
    predictor_tables <- map(predictors, function(cur_predictor) { 
      
            # Handle cases with and without predictor variables
        if (length(cur_predictor) > 0) {
          
          # Vector containing predictors and outcomes
          predictor_and_outcome <- c(cur_predictor, cur_outcome)
          base_defintion <- ifelse(is.null(base), NA, create_bases(base_info = base, variables = predictor_and_outcome)) # edit
          
          one_predictor_table <- data %>%
            # Filter missing values
            filter(rowSums(across(all_of(predictor_and_outcome), ~ is.na(.))) == 0) %>%
            # Group data
            group_by(across(all_of(cur_predictor))) %>% 
            # Weighted mean + base sizes, with grouping vars
            summarise(estimate = survey_mean(!!sym(cur_outcome), vartype = conf), 
                      base = unweighted(n())) %>% 
            # Added for consistency when merged with categorical outcome tables. 
            mutate("{cur_outcome}" := "w_mean") %>% 
            # Standardise names so can stack results across different predictors and outcomes. 
            standardise_names(data = data, preds = cur_predictor, out_var = cur_outcome) %>%
            mutate(cross_break = paste0(cur_predictor, collapse = "_X_")) %>%
            # edit
            mutate(base_description = base_defintion) # descriptor label for base 
        
        }
      
      return(one_predictor_table)
      
    })
    
    predictor_tables <- bind_rows(predictor_tables)
    
    } # Predictors ends
    
    # Bind crossbreaks and frequency table together
    if(!is.null(predictors)){table_for_one_outcome <- bind_rows(one_total_table, predictor_tables)
    } else {table_for_one_outcome <- one_total_table}
    
    table_for_one_outcome <- table_for_one_outcome %>%
      # Record statistic calculated
      mutate(stat = "w_mean") %>%
      # Add 'Total' to predictor category variables for frequency section of the table.
      mutate(across(contains(c("cross_break", "predictor", "p_lab", "p_cat")), ~case_when(is.na(.x) ~ "Total", TRUE ~ .x))) %>%
      # Reorder variables
      select(contains(c("cross_break", "predictor", "p_lab", "p_cat")), outcome, o_lab, o_cat, stat, contains("estimate"), base, base_description) # edit
    
    if(!is.null(conf)){
    if(conf == "ci"){
      table_for_one_outcome <- table_for_one_outcome %>% 
        mutate(estimate_ci = paste0(as.character(estimate_low), " - ", as.character(estimate_upp)))}
    } else { 
      table_for_one_outcome <- table_for_one_outcome %>% mutate(estimate_se = "-", .after = estimate)
      }

    return(table_for_one_outcome)
    
  }) # Outcomes loop ends 
  
  output <- output %>% bind_rows() %>% mutate(unweighted_n = base)
  
  return(output)
  
})})
}

write_function(weighted_mean, folder = "Scripts")

    if(!is.null(predictors)){
