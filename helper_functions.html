standardise_names <- 
  function(
    data, # dataframe used to create the table in question
    out_var, # outcome variable
    preds, # Grouping predictor variables
    table # Table containing variables to be standardised 
    )    {
    
    if(any(str_detect(class(data), "survey"))){data_to_look_in <- data[["variables"]]
    } else {data_to_look_in <- data}
        
    # Create a standardised set of variable names
    standardised_variable_names <- c(paste0("p_cat", seq_along(preds)), "o_cat")

    renamed_variables <- table %>%
          # Rename df columns with standard names
          rename(any_of(setNames(c(preds,out_var), standardised_variable_names)))  %>% 
          # Add outcome variable label
          mutate(o_lab = attributes(data_to_look_in[[out_var]])$label) %>%
          # Add column with outcome variable name
          mutate(outcome = out_var) 

        # Adding predictor meta data
        all_renamed <- map(seq_len(length(preds)), function(number) {

          # for each predictor in turn - create standardised variable names
          standardised_predictor_names <- paste0(c("predictor", "p_lab"), number)

          # Extract predictor variable labels
          p_label <- attributes(data_to_look_in[[preds[number]]])$label

          # Handle missing variable labels - not been required
          if(is.null(p_label)){p_label <- preds[number]}

          # Add predictor variable names and labels to ctab
          ctab_plus_new_labels <- renamed_variables %>%
            mutate(predictor = preds[number],
                   p_lab = p_label) %>%
            # Rename predictor variables with standardised names
            rename(any_of(setNames(c("predictor", "p_lab"), standardised_predictor_names)))

          return(ctab_plus_new_labels)

           }) %>% reduce(., full_join)
    
    return(all_renamed)
        
  }

write_function(standardise_names, folder = "Scripts")

4.3 Base descriptions

To work with base descriptions three functions are required:

  1. base_information - which creates a library of base descriptions for use when creating tables.

  2. create_bases - which uses the output of base_information to know what base descriptions to put with each table.

  3. prepare_base_for_table - takes the output of create_bases, stored within the resulting data frame, and creates a more human-readable output when formatting the resulting tables for presentation.


4.3.1 Function to create base descriptions

Purpose - creates an object you need in your environment to create tables with base sizes - it contains the variable base descriptions to use. The output is taken as an argument by functions for calculating statistics used later in the process.

Arguments.

  • data - the data frame containing the variables.
  • general_base - a character vector. A general base description for use with most variables.
  • specific_bases - a named character vector. Names are variable names. Content is a base description.

Where a variable has now corresponding description in specific_bases then general_base is used.

base_information <- function(
    data, 
    general_base, 
    specific_bases
) {
  
  everything_else <- general_base
  var_descriptions <- specific_bases
  
  variable_labels <- data %>% select(names(var_descriptions)) %>% var_label(unlist = T, null_action = "fill")
  all_variable_labels <- data %>% var_label(unlist = T, null_action = "fill")
  
  base_descriptions <- list(everything_else, var_descriptions, variable_labels, all_variable_labels)
  return(base_descriptions)
  
}

write_function(base_information, folder = "Scripts")

4.3.2 Create base info within tables

Purpose: This is the function used within larger functions calculating statistics (mean)

create_bases takes two arguments:

  • base_info is the output of the base_information() function.
  • variables is a character vector of variable names.

It exports a character vector which contains the base description for each variable name. If base descriptions are repeated, at this point they are simply duplicated in the output vector. Where there are multiple variables, their descriptions are separated by ” X “.

create_bases <- function(
    base_info,
    variables
){
  
  # Extract the names of variables that have specific bases defined (from base_info[[2]])
  names_of_variables_with_specific_bases <- base_info[[2]] %>% names
  
  # Logical vector: TRUE if variable has a specific base and is in 'variables', FALSE otherwise
  bases_to_use <- names_of_variables_with_specific_bases %in% variables
    
  # Get the labels of the variables that match bases_to_use (from base_info[[3]])
  variable_labels_to_use <- base_info[[3]][bases_to_use]
  
  # Get the specific bases that match bases_to_use (from base_info[[2]])
  bases_to_use <- base_info[[2]][bases_to_use] 
  
  # If there are any variables with specific bases...
  if(length(bases_to_use) > 0){
    # Create base strings in the form "<base>:-  <label>"
    list_of_bases <- paste0(bases_to_use, ":-  ", variable_labels_to_use)
  } else {
    # If no specific bases apply, start with an empty vector
    list_of_bases <- c()
  }
    
  # Find variables in 'variables' that do NOT have specific bases
  variables_using_default_description <- variables[!variables %in% names_of_variables_with_specific_bases]
  
  # If there are any variables without specific bases...
  if(length(variables_using_default_description) > 0){
    # Get their labels (from base_info[[4]])
    labels_of_variables_using_default_description <- base_info[[4]][variables_using_default_description]
    
    # Create default base strings in the form "<general_base>:-  <label>"
    default_base_description <- paste0(base_info[[1]],  ":-  ", labels_of_variables_using_default_description)
    
    # Append the default descriptions to the list_of_bases
    list_of_bases <- c(list_of_bases, default_base_description)
    } 
  
  # Collapse all base strings into one single string separated by " X "
  list_of_bases <- list_of_bases %>% paste0(., collapse = " X ")
  
  return(list_of_bases) # Return the final string

}

write_function(create_bases, folder = "Scripts")

4.3.3 Create formatted bases for display under a table

Purpose: This takes the long string of base descriptions returned by create_bases and makes into a human-readable piece of text.

Why not do this all in one go, as part of create_bases? Because, this returns a description for all variables in a table, and sometimes, the output of calc_stats, the function create_bases is used within, is later broken up into sub-tables. This stage happens after that, once the final variables to be shown in a table is known.

Arguments: it has only one argument, bases_for_table, which should be the character vector returned by create_bases.

Outputs: the outputs come as a list, each element corresponding to a different base description.

prepare_base_for_table <- function(bases_for_table) {
  
  base_results <- list()  # Empty list to store results

  # Split the input string by " X " into separate base elements, then keep only unique elements
  bases <- bases_for_table %>% str_split_1(., " X ") %>% unique
  
  # From each base element, extract the part before ":-  " as base description
  base_descriptions <- bases %>% str_split_i(., ":-  ", 1)
  
  # From each base element, extract the part after ":-  " as variable label(s)
  variables <- bases %>% str_split_i(., ":-  ", 2)
  
  # Find all unique base descriptions
  unique_base_descriptions <- unique(base_descriptions)
  
  # If there is only one unique base description...
  if(length(unique_base_descriptions) == 1){
    # Append that unique base description to the results list
    base_results <- append(base_results ,list(unique_base_descriptions))
    } else {
       
  # Create a data frame of `base_descriptions` and count the occurrences of each
  most_common_base_description <- data.frame(base_descriptions = base_descriptions) %>% 
    count(base_descriptions) %>%   # Count frequency of each base description
    mutate(max_n = max(n)) %>%     # Add a column with max count value
    filter(n == max_n) %>%         # Filter to base description(s) with max count
    pull(base_descriptions) %>%    # Extract the base description(s)
    unique                         # Remove duplicates if any
  
  # If there are multiple most common base descriptions, keep only the first one
  if(length(most_common_base_description) > 1){
    most_common_base_description <- most_common_base_description[1]
    }
  
  # Format the most common base description prepended by "All other variables:- " 
  base_results[[1]] <- most_common_base_description %>% 
    paste("All other variables:- ", ., collapse = "") %>% 
    str_to_sentence() # and capitalize sentence
  
  # Remove the most common base description from the unique list to handle separately
  unique_base_descriptions <- unique_base_descriptions[!unique_base_descriptions %in% most_common_base_description]
  
  # Loop over remaining unique base descriptions
  for (xxx in unique_base_descriptions){
    
    # Find indices where the base description matches the current unique description
    index <- which(base_descriptions %in% xxx) 
    
    # Concatenate the variables associated with this base description into a comma-separated string
    # `index` identifies the relevant variable labels from `variables`
    variable_list <- paste0(variables[index], collapse = ", ") 
    
    # Create a string combining the base description and its variables
    variable_description <- paste0(xxx, ":- ", variable_list) 
    
    # Append this description to the results list
    base_results <- append(base_results, variable_description)
    
  }
  
  }

    # Most common last. 
    base_results <- rev(base_results)
    
  # Return the list of prepared base descriptions
    return(base_results)

}

unweighted_test_numeric_by_cat <- function(data, outcome, predictor) {
  
  x <- data %>% pull(predictor)
  y <- data %>% pull(outcome)
  
  x_lab <- data %>% select(all_of(predictor)) %>% var_label(unlist = TRUE, null_action = "fill")
  y_lab <- data %>% select(all_of(outcome)) %>% var_label(unlist = TRUE, null_action = "fill")

  # Remove missing values
  complete_idx <- complete.cases(y, x)
  y <- y[complete_idx]
  x <- x[complete_idx]
  
  result <- tryCatch({
    # Check normality of residuals (Shapiro-Wilk)
    model <- lm(y ~ x)
    shapiro_p <- shapiro.test(residuals(model))$p.value
    
    # Decision
    if (shapiro_p < 0.05) {
      # Non-normal → Kruskal–Wallis
      method_used <- "Kruskal–Wallis"
      p_val <- kruskal.test(y ~ x)$p.value
    } else {
      # Normal enough → Welch's ANOVA
      method_used <- "Welch's ANOVA"
      p_val <- oneway.test(y ~ x, var.equal = FALSE)$p.value
    }
    
    list(method = method_used, p = p_val)
    
  }, error = function(e) {
    list(method = "Significance test returned an error.", p = NA)
  })
  
  result <- tibble(
    cross_break = predictor,
    predictor1 = predictor, 
    p_lab1 = x_lab, 
    outcome = outcome, 
    o_lab = y_lab,
    p_method = result$method,
    p_value = result$p
  )
  
  result <- map_df(c("mean", "median", "sum"), ~result %>% mutate(stat = .x))  
  
  return(result)
  
}

weighted_test_cat_by_cat <- function(data, outcome, predictor) {
  
  x_lab <- data[["variables"]] %>% select(all_of(predictor)) %>% var_label(unlist = TRUE, null_action = "fill")
  y_lab <- data[["variables"]] %>% select(all_of(outcome)) %>% var_label(unlist = TRUE, null_action = "fill")

  # Remove missing values
  filtered_data <- data %>% filter(if_all(c(outcome, predictor), ~!is.na(.x)))
  frmla <- formula(paste0("~",outcome,  " + " , predictor))
  
  result <- tryCatch({
    
      # chi-square test
      method_used <- "Chi-Square test"
      p_val <- svychisq(frmla, design = filtered_data, statistic = "adjWald")$p.value
      p_val <- p_val %>% as.vector()
      
    list(method = method_used, p = p_val)
    
  }, error = function(e) {
    list(method = "Significance test returned an error.", p = NA)
  })
  
  result <- tibble(
    cross_break = predictor,
    predictor1 = predictor, 
    p_lab1 = x_lab, 
    outcome = outcome, 
    o_lab = y_lab,
    p_method = result$method,
    p_value = result$p
  )

  result <- map_df(c("w_perc", "w_count"), ~result %>% mutate(stat = .x))  
  
  return(result)
  
}

return_pvalues <- function(data, outcome, predictor){

    if(any(str_detect(class(data), "survey"))){
      data_2 <- data[["variables"]]
      weights <- "weights"
    } else {
        data_2 <- data
        weights <- "no weights"
    }
  
  outcome_class_factor <- class(data_2 %>% pull(outcome)) %>% str_detect(., "factor") %>% any()
  
  if(weights == "no weights"){
    
    if(outcome_class_factor == TRUE){
      result <- unweighted_test_cat_by_cat(data = data, outcome = outcome, predictor = predictor)
    } else { 
      result <- unweighted_test_numeric_by_cat(data = data, outcome = outcome, predictor = predictor)
    }
    
  } else {
    
    if(outcome_class_factor == TRUE){
      result <- weighted_test_cat_by_cat(data = data, outcome = outcome, predictor = predictor) 
    } else { 
      result <- weighted_test_numeric_by_cat(data = data, outcome = outcome, predictor = predictor)
    }
    
  }
  
  return(result)

}
