df_to_dense_flextable <- function(
    df, 
    font_size = 8, 
    padding = 2, 
    merge = NULL, 
    slice = NULL
    ) {
  
  if(!is.null(slice)){df <- df %>% slice(slice)}
  
  ht <- df %>% as_hux %>% 
    # set_number_format(value =  list(function(x) formatC(x, digits = 1, format = "f"))) %>% 
    map_background_color(by_rows("#f5f7fa", "#ffffff")) %>% 
    huxtable::set_background_color(1, everywhere, "grey95") %>% 
    set_wrap(TRUE)
  
  if(!is.null(merge)){ht <- ht %>% merge_repeated_rows() %>% set_valign(value = "middle")}
  
  ft <- huxtable::as_flextable(ht) |>
    flextable::fontsize(size = font_size, part = "all") |>
    flextable::padding(padding = padding, part = "all") |>
    flextable::theme_booktabs() |>
    flextable::autofit()
  
  if(!is.null(merge)){ft <- ft %>% border(
    border = officer::fp_border(color = "grey"),
    # border.top = officer::fp_border(color = "grey"),
    # border.bottom = officer::fp_border(color = "grey"), 
    part = "body"
    )}
  
  format_numeric_cells <- function(ft, data) {
    flextable::set_formatter(
      ft,
      .formatter = function(x) {
        if (is.numeric(x)) {
          formatC(x, format = "f", digits = 1)
        } else {
          as.character(x)
        }
      }
    )
  }
  
  ft <- format_numeric_cells(ft, df)
  
  return(ft)
}

2.4 Function to save functions

write_function <- function(func_name, folder) {
  
  # Get the name of the function as a string
  func_name_str <- deparse(substitute(func_name))
  
  # Build the assignment line
  assign_statement <- paste0(func_name_str, " <- ")
  
  # Build the file path
  file_name <- file.path(folder, paste0(func_name_str, ".R"))
  
  # Get the function object itself
  function_to_export <- get(func_name_str)
  
  # Write to file
  writeLines(
    c(assign_statement, deparse(function_to_export)),
    con = file_name
  )
}

write_function(df_to_dense_flextable, folder = "Scripts")

source_folder_r <- function(folder, verbose = FALSE) {

  r_files <- list.files(folder, pattern = "\\.R$", full.names = TRUE)

  if (verbose) {
    cat("Sourcing files:\n")
    print(basename(r_files))
  }

  invisible(lapply(r_files, source))
}
